public class Solution {
 public void merge(int A[], int m, int B[], int n) {  
 //思路采用后插法，可以避免移动数目过多
          
        int i=m-1;  
        int j=n-1;  
        int k=m+n-1;  
          
        while(i>=0 && j>=0) {  
            A[k--] = A[i] > B[j] ? A[i--] : B[j--]; 
          //若i先结束才需要考虑，J先结束无需任何操作
        while(j>=0) {  
            A[k--] = B[j--];  
        }  
    }  
	
}
